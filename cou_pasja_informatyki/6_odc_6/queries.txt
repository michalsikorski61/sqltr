SELECT Continent, COUNT(Code) AS IlePanstw FROM country GROUP BY Continent #number of countries in continent - group by continent and in every continent count countries (and maybe order by IlePanstw)
SELECT Continent, COUNT(Code) AS IlePanstw FROM country GROUP BY Continent ORDER BY IlePanstw DESC #the same but with order by 
SELECT countrylanguage.Language, COUNT(countrylanguage.Language) AS IlePanstw FROM countrylanguage,country WHERE IsOfficial='T' AND country.Continent='Europe' AND countrylanguage.CountryCode=country.Code GROUP BY Language ORDER BY IlePanstw DESC; # how many countries use each language in Europe
SELECT country.Name, SUM(city.Population) AS IluObywateliPowyzej100k FROM city, country WHERE city.Population>=100000 AND country.Continent='South America' AND country.Code=city.CountryCode GROUP BY country.Name ORDER BY IluObywateliPowyzej100k DESC # 
SELECT gatunki.nazwa_zwyczajowa, COUNT(obserwacje.poczatek) AS IleObserwacji FROM gatunki, lokalizacje, obserwacje WHERE lokalizacje.powiat = 'gdanski' AND obserwacje.ID_lokalizacji=lokalizacje.ID_lokalizacji AND obserwacje.ID_gatunku=gatunki.ID_gatunku GROUP BY gatunki.nazwa_zwyczajowa ORDER BY IleObserwacji DESC LIMIT 10 # how many obserwations for every spiece was in district gdanski limit to 10 the largest numbers of observations
